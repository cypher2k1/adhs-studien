 //
 // Beispiel für eine if...else-Anweisung:
 // Der Variablen a wird der Wert 1 zugewiesen
 //
 a = 1;

 //
 // Im if-Zweig der Anweisung wird geprüft,
 // ob a den Wert 2 hat. Da die Bedingung nicht
 // erfüllt ist, wird die Anweisung nicht
 // ausgeführt.
 //
 if (a == 2)
     System.out.println ("a hat den Wert 2");  
  
 
 //
 // Stattdessen wird der else-Zweig ausgeführt.
 //
 else
     System.out.println ("a hat nicht den Wert 2");
