 //
 // Beispiel für eine while-Schleife
 // Die Variable i wird deklariert und
 // erhält den Wert 1.
 //
 int    i = 1;

 //
 // Die folgende Anweisung wird solange
 // ausgeführt, wie i kleiner oder
 // gleich 10 ist.
 //
 while (i <= 10) {

       System.out.println (i);  

 //
 // Jeweils am Ende der Anweisung wird der Wert
 // von i um 1 erhöht. Andernfalls würde
 // hier eine "Endlosschleife" entstehen,
 // die immer wieder den gleichen Wert ausgibt.
 //
       i ++;  
 }
